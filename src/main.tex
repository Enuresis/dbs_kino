\section{Špecifikácia}
\label{sec:specifikacia}

\subsection{Aký problém systém rieši a pre koho je určený?}
Systém rieši centralizovanú správu kina, konkrétne evidenciu filmov, plánovanie premietaní v sálach, predaj lístkov na konkrétne sedadlá a manažment úloh zamestnancov. Nahrádza manuálnu evidenciu, čím eliminuje duplicitu predaných miest a kolízie v harmonograme sál.
Systém je určený primárne pre manažéra kina (administrácia) a radových zamestnancov (predaj, údržba). Pridanou hodnotou je automatizácia kontroly kapacít a zefektívnenie prideľovania práce.
\subsection{Kto sú aktéri systému?}
\begin{itemize}
    \item \textbf{Manažér:} Má plné práva. Spravuje číselníky (filmy, sály), vytvára harmonogram premietaní a prideľuje pracovné úlohy zamestnancom.
    \item \textbf{Zamestnanec (Pokladník / Uvádzač):} Realizuje predaj lístkov a prehliada si zoznam svojich pridelených úloh.
    \item \textbf{Zákazník:} Neregistrovaný používateľ, ktorý si prezerá program a kupuje lístky.
\end{itemize}

\subsection{Aké sú kľúčové entity?}
\begin{itemize}
    \item \textbf{Film:} Základná entita nesúca informácie o diele (názov, dĺžka, žáner, vekové obmedzenie).
    \item \textbf{Kinosála:} Reprezentuje fyzickú miestnosť. Má definovaný typ (2D, 3D) a väzbu na sedadlá.
    \item \textbf{Sedadlo:} Konkrétne miesto v sále (rad, číslo).
    \item \textbf{Premietanie:} Spája Film a Kinosálu v konkrétnom čase.
    \item \textbf{Lístok:} Záznam o predaji. Viaže sa na jedno Premietanie a jedno konkrétne Sedadlo.
    \item \textbf{Zamestnanec:} Osoba v pracovnom pomere. Obsahuje atribúty ako typ úväzku a platnosť zdravotného preukazu.
    \item \textbf{Úloha:} Konkrétna pracovná činnosť pridelená zamestnancovi v určitom čase (napr. upratovanie sály po premietaní).
\end{itemize}

\subsection{Biznis pravidlá a integritné obmedzenia}

\begin{enumerate}
    \item \textbf{Unikátnosť lístka:} Dva platné lístky nemôžu odkazovať na rovnaké sedadlo v rámci toho istého premietania.
    \item \textbf{Časová exkluzivita sály:} V jednej kinosále nemôžu prebiehať dve premietania súčasne.
    \item \textbf{Kapacita sály:} Počet lístkov na premietanie nesmie prekročiť počet existujúcich sedadiel sály.
    \item \textbf{Oprávnenie zamestnanca:} Zamestnanec môže vykonávať úlohy s jedlom len vtedy, ak má v systéme evidovaný platný zdravotný preukaz.
    \item \textbf{Dostupnosť zamestnanca:} Jeden zamestnanec nemôže mať pridelené dve časovo sa prekrývajúce úlohy.
    \item \textbf{Konzistencia sedadiel:} Lístok je možné vytvoriť len na sedadlo, ktoré fyzicky patrí do kinosály, v ktorej sa koná dané premietanie.
\end{enumerate}

\subsection*{Scenár 1: Naplánovanie harmonogramu premietania}
\begin{itemize}
    \item \textbf{Názov a stručný popis:} Vytvorenie inštancie entity \textit{Premietanie}. Manažér priraďuje \textit{Film} do zvolenej \textit{Kinosály} na určený časový interval, pričom sa validuje časová exkluzivita sály.
    \item \textbf{Hlavný tok:}
    \begin{enumerate}
        \item Manažér inicializuje proces tvorby nového premietania a vyberie záznam z číselníka filmov.
        \item Systém načíta atribút \textit{dĺžka} z vybraného filmu.
        \item Manažér definuje \textit{čas\_začiatku} a vyberie cieľovú kinosálu.
        \item Systém algoritmicky vypočíta \textit{čas\_konca} (čas začiatku + dĺžka filmu + definovaná konštanta pre údržbu/upratovanie sály).
        \item Systém vykoná dopyt na existujúce premietania pre danú kinosálu a overí, či nedochádza k prieniku časových intervalov.
        \item Transakcia je potvrdená (commit) a záznam je perzistentne uložený.
    \end{enumerate}
    \item \textbf{Dátová vrstva}
    \begin{itemize}
        \item \textit{Čítanie:} \texttt{SELECT} nad tabuľkou \texttt{Film} pre získanie dĺžky. Následne \texttt{SELECT} nad tabuľkou \texttt{Premietanie} s vyhodnotením prekrývania časových pečiatok (\texttt{DATETIME}) pre daný \texttt{FK\_Kinosala}.
        \item \textit{Zápis:} \texttt{INSERT} do tabuľky \texttt{Premietanie}.
        \item \textit{Kontrola pravidiel:} Uplatňuje sa biznis pravidlo č. 2.
    \end{itemize}
    \item \textbf{Alternatívne toky:} Ak databázový dopyt vráti existujúci záznam v kolidujúcom čase, systém vyvolá výnimku, zruší zápis a vyzve manažéra na úpravu vstupných parametrov.
    \item \textbf{UML diagram:}
	    \begin{figure}[H]
	        \centering
	        \includegraphics[width=0.8\textwidth]{../assets/sc1_diag}
	        \caption{Diagram aktivít - Naplánovanie premietania}
	    \end{figure}
\end{itemize}

\subsection*{Scenár 2: Transakčný predaj lístkov s alokáciou sedadiel}
\begin{itemize}
    \item \textbf{Názov a stručný popis:} Pokladník realizuje predaj lístkov. Vyžaduje sa načítanie mapy sály, kontrola dostupnosti a vytvorenie záznamov rešpektujúcich unikátnosť sedadla.
    \item \textbf{Hlavný tok:}
    \begin{enumerate}
        \item Pokladník vyberie inštanciu aktuálneho premietania.
        \item Systém načíta topológiu sály a vyhodnotí obsadenosť jednotlivých sedadiel.
        \item Pokladník označí množinu voľných sedadiel (napr. 2 lístky) a inicializuje platbu.
        \item Systém bezprostredne pred zápisom overí aktuálnosť stavu.
        \item Systém vygeneruje záznamy pre lístky a označí ich ako predané.
    \end{enumerate}
    \item \textbf{Dátová vrstva}
    \begin{itemize}
        \item \textit{Čítanie:} Relačné spojenie (\texttt{JOIN}) tabuliek \texttt{Kinosala} a \texttt{Sedadlo}. Na zistenie voľných miest sa aplikuje \texttt{LEFT JOIN} na tabuľku \texttt{Listok} (záznamy s hodnotou \texttt{NULL} reprezentujú voľné sedadlá).
        \item \textit{Zápis:} Viacnásobný \texttt{INSERT} do tabuľky \texttt{Listok}.
        \item \textit{Kontrola pravidiel:} Uplatňuje sa integritné obmedzenie \texttt{UNIQUE(FK\_Premietanie, FK\_Sedadlo)} na databázovej úrovni, čo garantuje pravidlo č. 1 (Unikátnosť lístka). Zabezpečuje sa aj pravidlo č. 6 (Konzistencia sedadiel).
    \end{itemize}
    \item \textbf{Alternatívne toky:} Ak v čase medzi načítaním mapy a potvrdením iný používateľ obsadí vybrané sedadlo (porušenie \texttt{UNIQUE}), transakcia sa roll-backne a UI sa aktualizuje.
    \item \textbf{UML diagram:}

	        \includegraphics[width=0.8\textwidth]{../assets/sc2_diag}
	        \caption{Sekvenčný diagram - Predaj listkov a alokácia sedadiel}
	    \end{figure}
\end{itemize}

\subsection*{Scenár 3: Alokácia personálnych kapacít na prevádzkové úlohy}
\begin{itemize}
    \item \textbf{Názov a stručný popis:} Manažér priraďuje inštanciu entity \textit{Úloha} zamestnancovi. Systém filtruje zamestnancov na základe kvalifikačných obmedzení a časovej disponibility.
    \item \textbf{Hlavný tok:}
    \begin{enumerate}
        \item Manažér definuje typ úlohy a vyžadovaný časový interval.
        \item Systém načíta a aplikuje filtre na záznamy v tabuľke zamestnancov na základe typu úlohy.
        \item Zobrazí sa podmnožina zamestnancov, ktorí spĺňajú podmienky.
        \item Manažér vyberie zamestnanca a potvrdí alokáciu.
        \item Systém verifikuje, či zamestnanec nemá v danom čase priradenú inú úlohu.
        \item Systém vytvorí asociáciu medzi zamestnancom a úlohou.
    \end{enumerate}
    \item \textbf{Dátova vrstva}
    \begin{itemize}
        \item \textit{Čítanie:} Filtrovací \texttt{SELECT} nad entitou \texttt{Zamestnanec} (\texttt{WHERE platnost\_zdrav\_preukazu >= CURRENT\_DATE}). Dopyt nad tabuľkou \texttt{Uloha} overuje prienik intervalov pre dané \texttt{id\_zamestnanec}.
        \item \textit{Zápis:} \texttt{INSERT} do tabuľky \texttt{Uloha}.
        \item \textit{Kontrola pravidiel:} Uplatnenie biznis pravidiel č. 4 (Oprávnenie) a č. 5 (Dostupnosť zamestnanca).
    \end{itemize}
    \item \textbf{Alternatívne toky:} Ak neexistuje dostupný a kvalifikovaný zamestnanec, systém vráti prázdny result set a informuje manažéra.
    \item \textbf{UML diagram:}

	        \caption{Diagram aktivít - Alokácia presonálnych kapacít na prevádzkové úlohy}

\end{itemize}

\subsection*{Scenár 4: Generovanie agregovaných štatistík predajnosti}
\begin{itemize}
    \item \textbf{Názov a stručný popis:} Analytický dopyt agregujúci transakčné dáta za určité obdobie so zoskupením podľa filmových titulov.
    \item \textbf{Hlavný tok:}
    \begin{enumerate}
        \item Manažér si vyžiada štatistický prehľad a definuje časový parameter.
        \item Systém spustí analytický dopyt.
        \item Databázový stroj prepojí filmy, premietania a predané lístky.
        \item Systém vypočíta celkový počet predaných lístkov a tržby.
        \item Dataset je naformátovaný do prehľadnej tabuľky.

    \item \textbf{Dátova vrstva}

    \end{itemize}
    \item \textbf{Alternatívne toky:} Ak sa nerealizovalo žiadne premietanie, dopyt vráti prázdnu množinu a UI informuje o nulovej návštevnosti.

\section{Návrh relačného modelu}
\label{sec:relacny_model}

Navrhnutý fyzický dátový model je v 3. normálnej forme (3NF). Tabuľky využívajú umelý primárny kľúč (\texttt{id} typu \texttt{INT} s auto-inkrementom). M:N väzby sú plne dekomponované.

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{../assets/relac_model}
    \caption{Fyzický relačný model databázy kina v UML notácii}
    \label{fig:relacny_model}
\end{figure}

\subsection{Detailný popis tabuliek}

\subsubsection{Tabuľka: Film}
\begin{itemize}
    \item \textbf{Účel:} Centrálny katalóg filmových titulov.
    \item \textbf{Atribúty a obmedzenia:}
    \begin{itemize}
        \item \texttt{id (INT, PK)}.
        \item \texttt{nazov (VARCHAR, NOT NULL)}.
        \item \texttt{dlzka\_minut (INT, NOT NULL, CHECK > 0)}.
        \item \texttt{zaner (VARCHAR, NULL)}.
        \item \texttt{vekovy\_limit (INT, DEFAULT 0)}.
    \end{itemize}
\end{itemize}

\subsubsection{Tabuľka: Kinosala}
\begin{itemize}
    \item \textbf{Účel:} Evidencia priestorov určených na premietanie.
    \item \textbf{Atribúty a obmedzenia:}
    \begin{itemize}
        \item \texttt{id (INT, PK)}.
        \item \texttt{nazov (VARCHAR, NOT NULL, UNIQUE)}.
        \item \texttt{typ\_premietania (VARCHAR, CHECK IN ('2D','3D'))}.
    \end{itemize}
\end{itemize}

\subsubsection{Tabuľka: Sedadlo}
\begin{itemize}
    \item \textbf{Účel:} Priestorová matica pre konkrétnu kinosálu.
    \item \textbf{Atribúty a obmedzenia:}
    \begin{itemize}
        \item \texttt{id (INT, PK)}.
        \item \texttt{id\_kinosala (INT, FK)}.
        \item \texttt{rad (INT, NOT NULL, CHECK > 0)}, \texttt{cislo (INT, NOT NULL, CHECK > 0)}.
        \item \textbf{Integrita:} \texttt{UNIQUE(id\_kinosala, rad, cislo)}.
    \end{itemize}
\end{itemize}

\subsubsection{Tabuľka: Premietanie}
\begin{itemize}
    \item \textbf{Účel:} Prevádzkový harmonogram (spojenie filmu, sály a času).
    \item \textbf{Atribúty a obmedzenia:}
    \begin{itemize}
        \item \texttt{id (INT, PK)}.
        \item \texttt{id\_film (INT, FK)}, \texttt{id\_kinosala (INT, FK)}.
        \item \texttt{cas\_zaciatku (DATETIME, NOT NULL)}, \texttt{cas\_konca (DATETIME, NOT NULL)}.
    \end{itemize}
\end{itemize}

\subsubsection{Tabuľka: Listok}
\begin{itemize}
    \item \textbf{Účel:} Transakčná väzobná tabuľka reprezentujúca predaj.
    \item \textbf{Atribúty a obmedzenia:}
    \begin{itemize}
        \item \texttt{id (INT, PK)}.
        \item \texttt{id\_premietanie (INT, FK)}, \texttt{id\_sedadlo (INT, FK)}.
        \item \texttt{cena (DECIMAL(5,2), NOT NULL)}.
        \item \texttt{cas\_predaja (DATETIME, DEFAULT CURRENT\_TIMESTAMP)}.
        \item \textbf{Integrita:} \texttt{UNIQUE(id\_premietanie, id\_sedadlo)} – implementácia biznis pravidla č. 1 (Unikátnosť lístka).
    \end{itemize}
\end{itemize}

\subsubsection{Tabuľka: Zamestnanec}
\begin{itemize}
    \item \textbf{Účel:} Personálna evidencia.
    \item \textbf{Atribúty a obmedzenia:}
    \begin{itemize}
        \item \texttt{id (INT, PK)}.
        \item \texttt{meno (VARCHAR, NOT NULL)}, \texttt{priezvisko (VARCHAR, NOT NULL)}.
        \item \texttt{typ\_uvazku (VARCHAR, NOT NULL)}.
        \item \texttt{platnost\_zdrav\_preukazu (DATE, NULL)}.
    \end{itemize}
\end{itemize}

\subsubsection{Tabuľka: Uloha}
\begin{itemize}
    \item \textbf{Účel:} Harmonogram úloh priradených zamestnancom.
    \item \textbf{Atribúty a obmedzenia:}
    \begin{itemize}
        \item \texttt{id (INT, PK)}.
        \item \texttt{id\_zamestnanec (INT, FK)}.
        \item \texttt{id\_kinosala (INT, FK, NULL)}: Umožňuje viazať úlohu (napr. upratovanie, kontrola) na konkrétnu sálu. Hodnota NULL pre úlohy nezávislé od sály (napr. pokladňa).
        \item \texttt{typ\_ulohy (VARCHAR, NOT NULL)}.
        \item \texttt{cas\_zaciatku (DATETIME, NOT NULL)}, \texttt{cas\_konca (DATETIME, NOT NULL)}.
    \end{itemize}
\end{itemize}

\subsection{Popis vzťahov a referenčná integrita}

Všetky definované vzťahy sú typu \textbf{1:N}.

\begin{itemize}
    \item \textbf{Kinosala (1) --- (1..N) Sedadlo}: \texttt{ON DELETE CASCADE}. Odstránením záznamu kinosály zanikajú všetky k nej priradené sedadlá.
    \item \textbf{Film (1) --- (0..N) Premietanie}: \texttt{ON DELETE RESTRICT}. Zabráni vymazaniu filmu, ak na neho existuje naplánované premietanie.
    \item \textbf{Kinosala (1) --- (0..N) Premietanie}: \texttt{ON DELETE RESTRICT}. Zabráni odstráneniu kinosály, v ktorej je naplánovaný program.
    \item \textbf{Premietanie (1) --- (0..N) Listok} a \textbf{Sedadlo (1) --- (0..N) Listok}: \texttt{ON DELETE RESTRICT}. Predstavuje finančný a transakčný záznam. Kým existuje aspoň jeden predaný lístok, rodičovské záznamy (premietanie ani sedadlo) nesmú byť vymazané.
    \item \textbf{Zamestnanec (1) --- (0..N) Uloha}: \texttt{ON DELETE CASCADE}. Pri výmaze zamestnanca zo systému (napr. GDPR požiadavka) sa odstraňuje aj jeho harmonogram úloh.
    \item \textbf{Kinosala (1) --- (0..N) Uloha}: \texttt{ON DELETE CASCADE}. Ak zanikne sála, zanikajú aj naplánované prevádzkové úlohy viazané exkluzívne na tento priestor.
\end{itemize}

\subsection{Zabezpečenie integritných obmedzení a pokrytie scenárov}
\begin{enumerate}
    \item \textbf{Unikátnosť lístka:} Implementované \textbf{na dátovej vrstve} kompozitným obmedzením \texttt{UNIQUE(id\_premietanie, id\_sedadlo)} v tabuľke \texttt{Listok}.
    \item \textbf{Časová exkluzivita sály:} Implementované \textbf{na aplikačnej vrstve} (Scenár 1). Overuje sa dopytom na prienik časových intervalov pred \texttt{INSERT} operáciou.
    \item \textbf{Kapacita sály:} Vyplýva \textbf{z fyzickej štruktúry dát}. Vzhľadom na \texttt{UNIQUE} index lístka nie je technicky možné predať viac lístkov, než existuje asociovaných záznamov v tabuľke \texttt{Sedadlo}.
    \item \textbf{Oprávnenie zamestnanca:} Implementované \textbf{na aplikačnej vrstve} (Scenár 3) pridaním filtra \texttt{WHERE platnost\_zdrav\_preukazu >= CURRENT\_DATE} pri alokácii práce s jedlom.
    \item \textbf{Dostupnosť zamestnanca:} Riešené \textbf{na aplikačnej vrstve} (Scenár 3) kontrolným dotazom na kolidujúce časy v tabuľke \texttt{Uloha}.
    \item \textbf{Konzistencia sedadiel:} Implementované \textbf{cez transakčnú logiku} (Scenár 2). Aplikácia načíta platnú topológiu sály (\texttt{JOIN} Kinosala-Sedadlo) výhradne pre sálu priradenú k danému premietaniu.
\end{enumerate}

\section{Zhodnotenie}
\label{sec:zhodnotenie}

\begin{itemize}
    \item \textbf{Ktorá časť riešenia vám zabrala najviac času a prečo?}
    Najviac času nám zabral samotný návrh tabuliek a vzťahov tak, aby dávali zmysel a pokrývali naše zadanie.

    \item \textbf{S čím ste si nevedeli poradiť? Ako ste to vyriešili?}
    Chvíľu sme bojovali s nastavením pravidiel mazania pre cudzie kľúče. Nevedeli sme sa na začiatku rozhodnúť, čo sa má zmazať spolu s rodičovskou tabuľkou a čo nie.

    \item \textbf{Čo bolo pre vás úplne nové?}
    Úplne nové bolo pre nás formálne modelovanie používateľských scenárov z pohľadu databázových operácií (definovanie konkrétnych \texttt{SELECT}, \texttt{INSERT} a \texttt{JOIN} krokov).

    \item \textbf{Pokrýva vaše riešenie správne zvolenú tému? Čo ste zjednodušili alebo vynechali?}
